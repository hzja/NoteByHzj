import requests
from requests.packages.urllib3.exceptions import InsecureRequestWarning
requests.packages.urllib3.disable_warnings(InsecureRequestWarning)

def scan(url):
    with open("D:/漏洞扫描器/xray/new 12.txt","r") as f: #用以读方式打开e盘目录下的txt文件
        for i in f.readlines():  
            i = i.strip()   #去除空格
            urls = ('https://'+url)
            #print(urls)
            r = requests.get(urls+i,verify=False) #request发送请求如果状态码200证明存在url链接
            #print(r.status_code)
            if r.status_code == 200:
                print("url:"+r.url+"========================存在")
            else:
                #pass
                print("url:"+r.url+"不存在")

if __name__ == '__main__':
    print('''
    注意！输入要去除协议，例如https://baidu.com →baidu.com
    用于扫描中间件、spring的特殊漏洞路径。
    ''')
    url = input('请输入要扫描的网站: ')
    scan(url)
